2. 单链表反转

``` typescript
reverse() {
        let newList: nodeType = new Node(null);
        let current = this.head.next;
        let next: nodeType;
        while (current) {
            next = current.next; // 暂存当前节点的下一个节点,后面需要用到
            current.next = newList.next; // 当前current的下一个节点指向新的节点的最前端
            newList.next = current;// 将 current 赋值给新链表
            current = next //链表指针后移
        }
        this.head = newList;
    }
```
